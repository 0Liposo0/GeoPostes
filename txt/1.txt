import flet as ft

def main(page: ft.Page):
    page.title = 'Flet App'
    page.window.always_on_top = True # Manter a janela sempre no topo
    page.window.height = 960
    page.window.width = 540 
    page.window.resizable = False  
    page.bgcolor = ft.colors.WHITE  
    page.scroll = "auto"  # Adiciona a rolagem automática à página

    class Poste:
        def __init__(self, ip, situacao, tipo, pontos, bairro, logradouro):
            self.ip = ip
            self.situacao = situacao
            self.tipo = tipo
            self.pontos = pontos
            self.bairro = bairro
            self.logradouro = logradouro

    poste = Poste("IP CAP-0040", "Com iluminação", "Lâmpada LED", 1, "Centro", "Rua Raimundo Malta" )

    def chamada1(e):
        global poste
        poste1 = Poste("IP CAP-0001", "Com iluminação", "Lâmpada LED", 1, "Centro", "Rua Raimundo Malta" )
        poste = poste1
        forms.visible = True
        forms.update()
        foto.visible = True
        foto.update()
        ordem.visible = True
        ordem.update()
        titulo.visible = False
        titulo.update()
        mapa.visible = False
        mapa.update()
        texto_chamada.visible = False
        texto_chamada.update()
        page.scroll_to(9999)

    
    def restart(e):
        forms.visible = False
        forms.update()
        foto.visible = False
        foto.update()
        ordem.visible = False
        ordem.update()
        titulo.visible = True
        titulo.update()
        mapa.visible = True
        mapa.update()
        texto_chamada.visible = True
        texto_chamada.update()
    
    facens = ft.Image(
        data = 0,
        src = "images/facens.webp",
        repeat = ft.ImageRepeat.NO_REPEAT,
        height=170,
    )

    titulo = ft.Container(
        visible=True,
        col=12,
        padding=ft.padding.all(10),
        content=ft.Row(
            alignment=ft.MainAxisAlignment.CENTER,
            spacing=0,
            controls=[
                ft.Text(
                    value="GeoPostes",
                    color=ft.colors.BLACK,
                    size=50,
                    weight = ft.FontWeight.W_600,
                    font_family= "Tahoma", 
                    ),
                facens
                
            ]
        )
    )

    btn1 = ft.ElevatedButton(
        on_click=chamada1,
        width=20,  
        height=20,
        bgcolor=ft.colors.AMBER,
        text=" ",
        style= ft.ButtonStyle(
            shape=ft.RoundedRectangleBorder(radius=10),     
        )
    )

    btn2 = ft.ElevatedButton(
        on_click=chamada1,
        width=20,  
        height=20,
        bgcolor=ft.colors.AMBER,
        text=" ",
        style= ft.ButtonStyle(
            shape=ft.RoundedRectangleBorder(radius=10),     
        )
    )

    seta = ft.Image(
        data = 0,
        src = "images/seta.png",
        repeat = ft.ImageRepeat.NO_REPEAT,
        height=70,
    )


    mapa = ft.Column(
        visible=True, 
        col=12,
        horizontal_alignment=ft.CrossAxisAlignment.CENTER,
        controls= [ ft.Container(
            width=400, 
            height=400,
            alignment= ft.alignment.center, 
            image_src= "images/maps.jpg", 
            border_radius=ft.border_radius.all(250),
            content=ft.Column(
                spacing=20,
                alignment= ft.MainAxisAlignment.CENTER,
                controls=[ 
                ft.Row(
                    alignment= ft.MainAxisAlignment.CENTER,
                    spacing=80,
                    controls=[btn1, seta, btn2]
                ),

                ],
            )
        )      
        ]
    )

   
    theme1 =ft.Theme(
        text_theme= ft.TextTheme(
            title_large = ft.TextStyle(
                size=15,
                color = ft.colors.BLACK,
                weight = ft.FontWeight.W_900,                                   
            ),
            title_medium = ft.TextStyle(
                size=15,
                color = ft.colors.BLACK,
                weight = ft.FontWeight.W_400,                                   
            ),
        )
    )


    texto_chamada = ft.Container(
        visible=True,
        col=12,
        padding=20,
        content=(
            ft.Text(
                value="Aproxime-se e clique em um ponto",
                text_align= ft.TextAlign.CENTER,
                size=20,
                color = ft.colors.BLACK,
                weight = ft.FontWeight.W_900,
            )
        ),
        theme = theme1,

    )


    poste1 = ft.Image(
        data = 0,
        src = "images/poste1.jpg",
        repeat = ft.ImageRepeat.NO_REPEAT,
        height=400,
    )   
    

    forms =ft.Container( 
        col=12,
        visible=False,
        content=ft.DataTable(
            data_row_max_height=50,
            columns=[
                ft.DataColumn(ft.Text("")),
                ft.DataColumn(ft.Text("")),
            ],
             rows=[
                ft.DataRow(cells = [ft.DataCell(ft.Text(value="IP", style=ft.TextThemeStyle.TITLE_LARGE)),
                                    ft.DataCell(ft.Text(value=poste.ip, style=ft.TextThemeStyle.TITLE_MEDIUM)),],),
                ft.DataRow(cells = [ft.DataCell(ft.Text(value="Situação", style=ft.TextThemeStyle.TITLE_LARGE)),
                                    ft.DataCell(ft.Text(value=poste.situacao, style=ft.TextThemeStyle.TITLE_MEDIUM)),],),
                ft.DataRow(cells = [ft.DataCell(ft.Text(value="Tipo de Lâmpada", style=ft.TextThemeStyle.TITLE_LARGE)),
                                    ft.DataCell(ft.Text(value=poste.tipo, style=ft.TextThemeStyle.TITLE_MEDIUM)),],),
                ft.DataRow(cells = [ft.DataCell(ft.Text(value="Pontos", style=ft.TextThemeStyle.TITLE_LARGE)),
                                    ft.DataCell(ft.Text(value=poste.pontos, style=ft.TextThemeStyle.TITLE_MEDIUM)),],),
                ft.DataRow(cells = [ft.DataCell(ft.Text(value="Bairro", style=ft.TextThemeStyle.TITLE_LARGE)),
                                    ft.DataCell(ft.Text(value=poste.bairro, style=ft.TextThemeStyle.TITLE_MEDIUM)),],),
                ft.DataRow(cells = [ft.DataCell(ft.Text(value="Logradouro", style=ft.TextThemeStyle.TITLE_LARGE)),
                                    ft.DataCell(ft.Text(value=poste.logradouro, style=ft.TextThemeStyle.TITLE_MEDIUM)),],),
                ft.DataRow(cells = [ft.DataCell(ft.Text(value="", style=ft.TextThemeStyle.TITLE_LARGE)),
                                    ft.DataCell(ft.Text(value="", style=ft.TextThemeStyle.TITLE_MEDIUM)),],),
            ],
            ),
            theme = theme1,
            )
    
    foto=ft.Container(
        col=12,
        visible=False,
        content=ft.Column(
            alignment=ft.MainAxisAlignment.CENTER,
            horizontal_alignment=ft.CrossAxisAlignment.CENTER,
            controls=[
                ft.Text(value="Foto", style=ft.TextThemeStyle.TITLE_LARGE),
                poste1   
            ]
        ),
        theme = theme1,
    )

    ordem = ft.Container(
        visible=False,
        col=6,
        padding=20,
        content=(
            ft.ElevatedButton(
                text="Abrir chamado",
                bgcolor=ft.colors.GREEN,
                color = ft.colors.BLACK,
            )
        ),
    )

    page.floating_action_button = ft.FloatingActionButton(
        icon=ft.icons.CHANGE_CIRCLE,
        on_click=restart  
    )

    layout = ft.ResponsiveRow(
        columns=12,
        controls=[
            titulo,
            mapa,
            texto_chamada,
            forms,
            foto,
            ordem
        ],
        alignment=ft.MainAxisAlignment.CENTER,
        vertical_alignment=ft.CrossAxisAlignment.CENTER,
    )

    page.add(layout)


ft.app(target=main, assets_dir="assets")
