import flet as ft
from views import *
from models import *


def main(page: ft.Page):
    # Configurações da janela
    page.title = 'Flet App'
    page.window.always_on_top = True
    page.window.height = 900
    page.window.width = 450 
    page.window.resizable = False  
    page.bgcolor = ft.colors.WHITE  
    page.scroll = "auto"  # Habilitar rolagem automática na página
    
    # Estado inicial do poste
    poste = Poste("IP SOR-0040", "Com iluminação", "Lâmpada LED", 1, "Centro", "Rua Raimundo Malta")

    # Criação dos componentes visuais
    titulo = create_titulo()
    icon = create_icon()
    mapa = create_mapa(
        page,
        btn1_action=lambda e: page_forms(e, page, poste),
        btn2_action=lambda e: page_forms(e, page, poste)
    )

    # Botão flutuante para reiniciar a tela
    page.floating_action_button = ft.FloatingActionButton(
    icon=ft.icons.HOUSE,
    on_click=lambda e: restart(e, page, poste)  # Passa o evento e a página
)

    # Layout inicial
    layout = ft.ResponsiveRow(
        columns=12,
        controls=[
            titulo,
            mapa,
            texto_chamada,
            icon
        ],
        alignment=ft.MainAxisAlignment.CENTER,
        vertical_alignment=ft.CrossAxisAlignment.CENTER,
    )

    # Adiciona o layout à página
    page.add(layout)

# Executa a aplicação
if __name__ == "__main__":
    ft.app(target = main, assets_dir="assets")
