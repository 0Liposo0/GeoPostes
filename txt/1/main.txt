import flet as ft
from views import create_titulo, create_mapa, create_forms, create_foto, create_ordem 
from models import Poste, texto_chamada

def main(page: ft.Page):
    # Configurações da janela
    page.title = 'Flet App'
    page.window.always_on_top = True
    page.window.height = 960
    page.window.width = 540 
    page.window.resizable = False  
    page.bgcolor = ft.colors.WHITE  
    page.scroll = "auto"  # Habilitar rolagem automática na página
    
    # Estado inicial do poste
    poste = Poste("IP CAP-0040", "Com iluminação", "Lâmpada LED", 1, "Centro", "Rua Raimundo Malta")

    # Função para chamada 1 (atualiza a interface com informações específicas do poste)
    def chamada1(e):
        nonlocal poste
        poste = Poste("IP CAP-0001", "Com iluminação", "Lâmpada LED", 1, "Centro", "Rua Raimundo Malta")
        forms.visible = True
        forms.update()
        foto.visible = True
        foto.update()
        ordem.visible = True
        ordem.update()
        titulo.visible = False
        titulo.update()
        mapa.visible = False
        mapa.update()
        texto_chamada.visible = False
        texto_chamada.update()
        page.scroll_to(9999)  # Rola a página até o final

    # Função para reiniciar a interface para o estado original
    def restart(e):
        forms.visible = False
        forms.update()
        foto.visible = False
        foto.update()
        ordem.visible = False
        ordem.update()
        titulo.visible = True
        titulo.update()
        mapa.visible = True
        mapa.update()
        texto_chamada.visible = True
        texto_chamada.update()

    # Criação dos componentes visuais
    titulo = create_titulo()
    mapa = create_mapa(chamada1, chamada1)  # Atribui chamada1 para os dois botões
    forms = create_forms(poste)
    foto = create_foto()
    ordem = create_ordem()

    # Botão flutuante para reiniciar a tela
    page.floating_action_button = ft.FloatingActionButton(
        icon=ft.icons.CHANGE_CIRCLE,
        on_click=restart
    )

    # Layout responsivo
    layout = ft.ResponsiveRow(
        columns=12,
        controls=[
            titulo,
            mapa,
            texto_chamada,
            forms,
            foto,
            ordem
        ],
        alignment=ft.MainAxisAlignment.CENTER,
        vertical_alignment=ft.CrossAxisAlignment.CENTER,
    )

    # Adiciona o layout à página
    page.add(layout)

# Executa a aplicação
ft.app(target=main, assets_dir="assets")
